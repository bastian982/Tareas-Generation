function findCommonCourses(curso1, curso2) {
let cursoComun = [];
for (let i = 0; i < curso1.length; i++) {
let curso = curso1[i];
for (let j = 0; j < curso2.length; j++) {
if (curso === curso2[j]) {
cursoComun.push(curso);
break;
}
}
}
for (let i = 0; i < cursoComun.length; i++) {
console.log(`Cursos en comun: ${cursoComun[i]}`);
}
}
