function fncSumAndProd(numbers){
return numbers.reduce(function(a, b){
return{
sum: a.sum + b,
multi: a.multi * b };
}, {sum: 0, multi: 1});
}
const numbers = [1, 2, 3, 4];
const {sum, multi} = fncSumAndProd(numbers);
console.log(`La suma de ${numbers} es: ${sum}, la multiplicación de ${numbers} es: ${multi}` );
